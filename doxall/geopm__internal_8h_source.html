156 static inline uint64_t geopm_region_id_hint(uint64_t region_id)
157 {
158  uint64_t ret;
159  if (GEOPM_REGION_HASH_UNMARKED == region_id) {
160  ret = GEOPM_REGION_HINT_UNKNOWN;
161  }
162  else if (geopm_region_id_is_mpi(region_id)) {
163  ret = GEOPM_REGION_HINT_NETWORK;
164  }
165  else {
166  ret = region_id & GEOPM_MASK_REGION_HINT;
167  if (!ret) {
168  ret = GEOPM_REGION_HINT_UNKNOWN;
169  }
170  }
171  return ret;
