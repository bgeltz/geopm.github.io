172  template <class type>
173  void CircularBuffer<type>::set_capacity(const unsigned int size)
174  {
175  if (size < m_count && m_max_size > 0) {
176  int size_diff = m_count - size;
177  std::vector<type> temp;
178  //Copy newest data into temporary vector
179  for (unsigned int i = m_head + size_diff; i != ((m_head + m_count) % m_max_size); i = ((i + 1) % m_max_size)) {
180  temp.push_back(m_buffer[i]);
181  }
182  //now re-size and swap out with tmp vector data
183  m_buffer.resize(size);
184  m_buffer.swap(temp);
185  m_count = size;
186  }
187  else {
188  m_buffer.resize(size);
189  }
190  m_head = 0;
191  m_max_size = size;
192  }
