52 /* Macro seems to be the best way to deal introducing per function
53  * static storage with a non-const initializer. We avoid repeating
54  * code. Note this approach is thread safe because of the underlying
55  * lock in geopm_prof_region().
56  */
57 #define GEOPM_PMPI_ENTER_MACRO(FUNC) \
58  static unsigned is_once = 1; \
59  static uint64_t func_rid = 0; \
60  if (is_once || func_rid == 0) { \
61  func_rid = geopm_mpi_func_rid(FUNC); \
62  is_once = 0; \
63  } \
64  geopm_mpi_region_enter(func_rid);
