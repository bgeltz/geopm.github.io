184  template <class type>
185  void CircularBuffer<type>::set_capacity(const unsigned int size)
186  {
187  if (size < m_count) {
188  int size_diff = m_count - size;
189  std::vector<type> temp;
190  //Copy newest data into temporary vector
191  for (unsigned int i = m_head + size_diff; i < ((m_head + m_count) % m_max_size); i = ((i + 1) % m_max_size)) {
192  temp.push_back(m_buffer[i]);
193  }
194  //now re-size and swap out with tmp vector data
195  m_buffer.resize(size);
196  m_buffer.swap(temp);
197  m_count = size;
198  }
199  else {
200  m_buffer.resize(size);
201  }
202  m_head = 0;
203  m_max_size = size;
204  }

206  template <class type>
207  void CircularBuffer<type>::insert(const type value)
208  {
209  if (m_max_size < 1) {
210  throw Exception("CircularBuffer::insert(): Cannot insert into a buffer of 0 size", GEOPM_ERROR_RUNTIME, __FILE__, __LINE__);
211  }
212  if (m_count < m_max_size) {
213  m_buffer[m_count] = value;
214  m_count++;
215  }
216  else {
217  m_buffer[m_head] = value;
218  m_head = ((m_head + 1) % m_max_size);
219  }
220  }

233  {
234  std::vector<type> result(size());

236  std::copy(m_buffer.begin(), m_buffer.begin() + m_count, result.begin());

241  }
